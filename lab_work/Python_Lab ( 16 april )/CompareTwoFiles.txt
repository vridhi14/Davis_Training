# Algorithm:

# 1 Start
# 2 Open file1.txt and file2.txt
# 3 Read lines from both files
# 4 Store lines in sets
# 5 Find lines present in file1 but not in file2
# 6 Display those lines
# 7 Stop

with open("file1.txt") as f1, open("file2.txt") as f2:
    lines1 = set(f1.readlines())
    lines2 = set(f2.readlines())

difference = lines1 - lines2

for line in difference:
    print(line.strip())
